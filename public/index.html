        function addToTimeline(type) {

            if (type === 'museum') {

                // Shake effect

        function runChatDemo() {

            if (state.chatActive) return;

            state.chatActive = true;

            const chatWindow = document.getElementById('chat-window');

            const btn = document.getElementById('chat-btn');

            

            btn.disabled = true;

            btn.innerText = "הסוכן מעבד נתונים...";

            btn.className = "w-full py-2 bg-gray-100 text-gray-400 font-bold rounded-lg cursor-not-allowed";


            const responses = [

                { text: "בודק את זה מיד מול חברת ההסעות... 🤖", delay: 1000, side: 'left' },

                { text: "איתרתי את ההזמנה. עדכנתי את הנהג על העיכוב בטיסה LY324. 🚕", delay: 3000, side: 'left' },

                { text: "הנהג אישר. הוא ימתין לך בשער 4 בשעה 16:30. אין צורך בתשלום נוסף. תהנו! ✨", delay: 5500, side: 'left' }

            ];


            let totalDelay = 0;


            responses.forEach(resp => {

                totalDelay += resp.delay;

                setTimeout(() => {

                    const msgDiv = document.createElement('div');

                    msgDiv.className = `chat-message p-3 rounded-lg shadow-sm max-w-[85%] text-sm ${resp.side === 'left' ? 'bg-indigo-50 text-indigo-900 self-start ml-auto rounded-tl-none' : 'bg-white self-end mr-auto rounded-tr-none'}`;

                    msgDiv.innerText = resp.text;

                    chatWindow.appendChild(msgDiv);

                    chatWindow.scrollTop = chatWindow.scrollHeight;

                }, totalDelay);

            });


            setTimeout(() => {

                btn.innerText = "הבעיה נפתרה בהצלחה ✅";

                btn.className = "w-full py-2 bg-green-100 text-green-700 font-bold rounded-lg border border-green-200";

                state.chatActive = false;

            }, totalDelay + 500);

        }
